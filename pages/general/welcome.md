## Quick Example

```js
const { Client, Intents } = require("discord.js");
const { Moderation } = require("@djs-modules/moderation");

const client = new Client({
  intents: [
    "GUILDS",
    "GUILD_MESSAGES",
    "GUILD_PRESENCES",
    "GUILD_MEMBERS",
    "GUILD_BANS",
  ],
});

const moderation = new Moderation(client, {
  dbPath: "./",
  locale: "en-US",
  defaultSystems: {
    autoRole: false,
    antiSpam: false,
    antiInvite: false,
    antiJoin: false,
    antiLink: false,
    blacklist: false,
    ghostPing: false,
    logSystem: false,
  },
});
```
